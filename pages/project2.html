int* twoSum(int* nums, int numsSize, int target, int* returnSize){
    int i,j;
    int* p = (int*)malloc(sizeof(int)*2);
    for(i=0;i<numsSize;i++)
    {
        for(j=i+1;j<numsSize;j++)
        {
            if(nums[i] + nums[j] == target)
            {
                p[1] = i;
                p[0] = j;
            }
        }
    }
    *returnSize = 2;
    return p;
}

Problem #13


int romanToInt(char * s){

typedef struct {
    int I;
    int V;
    int X;
    int L;
    int C;
    int D;
    int M;
}Language;
Language roman = {1,5,10,50,100,500,1000};
int i,j,sum=0;
for(i=0;i<strlen(s);i++)
{
    switch(s[i])
    {
        case 'I': sum+=roman.I;
            break;
        case 'V': sum+=roman.V;
            break;
        case 'X': sum+=roman.X;
            break;
        case 'L': sum+=roman.L;
            break;
        case 'C': sum+=roman.C;
            break;
        case 'D': sum+=roman.D;
            break;
        case 'M': sum+=roman.M;
            break;
    }
}
for(i=1;i<strlen(s);i++)
{
    if(((s[i] == 'V') || (s[i] == 'X')) && (s[i-1] == 'I'))
    {
        sum-=2;
    }
    if(((s[i] == 'L') || (s[i] == 'C')) && (s[i-1] == 'X'))
    {
        sum-=20;
    }
    if(((s[i] == 'M') || (s[i] == 'D')) && (s[i-1] == 'C'))
    {
        sum-=200;
    }  
}
return sum;
}

Problem #914


class Solution {
    public:
        bool hasGroupsSizeX(vector<int>& deck) {
            if(deck.size() == 1)return false;
            unordered_map <int, int> map;
            int max=0, min=INT_MAX;
            for( int& n : deck)
            {
                map[n]++;
                if(max < map[n])
                    max = map[n];
            }
            for(auto &[a,b]:map)
                if(min>b)
                    min=b;
            
                for(auto &[a,b]:map)
                    if(b!=max and (__gcd(max, b) <= 1 or __gcd(min, b) <= 1))
                        return false;
            return true;
        }
};
